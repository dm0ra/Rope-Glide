165  private static void SetPositionVisibleinSceneView(RectTransform canvasRTransform, RectTransform itemTransform)
166  {
167  // Find the best scene view
168  SceneView sceneView = SceneView.lastActiveSceneView;
169  if (sceneView == null && SceneView.sceneViews.Count > 0)
170  sceneView = SceneView.sceneViews[0] as SceneView;
171 
172  // Couldn't find a SceneView. Don't set position.
173  if (sceneView == null || sceneView.camera == null)
174  return;
175 
176  // Create world space Plane from canvas position.
177  Vector2 localPlanePosition;
178  Camera camera = sceneView.camera;
179  Vector3 position = Vector3.zero;
180  if (RectTransformUtility.ScreenPointToLocalPointInRectangle(canvasRTransform, new Vector2(camera.pixelWidth / 2, camera.pixelHeight / 2), camera, out localPlanePosition))
181  {
182  // Adjust for canvas pivot
183  localPlanePosition.x = localPlanePosition.x + canvasRTransform.sizeDelta.x * canvasRTransform.pivot.x;
184  localPlanePosition.y = localPlanePosition.y + canvasRTransform.sizeDelta.y * canvasRTransform.pivot.y;
185 
186  localPlanePosition.x = Mathf.Clamp(localPlanePosition.x, 0, canvasRTransform.sizeDelta.x);
187  localPlanePosition.y = Mathf.Clamp(localPlanePosition.y, 0, canvasRTransform.sizeDelta.y);
188 
189  // Adjust for anchoring
190  position.x = localPlanePosition.x - canvasRTransform.sizeDelta.x * itemTransform.anchorMin.x;
191  position.y = localPlanePosition.y - canvasRTransform.sizeDelta.y * itemTransform.anchorMin.y;
192 
193  Vector3 minLocalPosition;
194  minLocalPosition.x = canvasRTransform.sizeDelta.x * (0 - canvasRTransform.pivot.x) + itemTransform.sizeDelta.x * itemTransform.pivot.x;
195  minLocalPosition.y = canvasRTransform.sizeDelta.y * (0 - canvasRTransform.pivot.y) + itemTransform.sizeDelta.y * itemTransform.pivot.y;
196 
197  Vector3 maxLocalPosition;
198  maxLocalPosition.x = canvasRTransform.sizeDelta.x * (1 - canvasRTransform.pivot.x) - itemTransform.sizeDelta.x * itemTransform.pivot.x;
199  maxLocalPosition.y = canvasRTransform.sizeDelta.y * (1 - canvasRTransform.pivot.y) - itemTransform.sizeDelta.y * itemTransform.pivot.y;
200 
201  position.x = Mathf.Clamp(position.x, minLocalPosition.x, maxLocalPosition.x);
202  position.y = Mathf.Clamp(position.y, minLocalPosition.y, maxLocalPosition.y);
203  }
204 
205  itemTransform.anchoredPosition = position;
206  itemTransform.localRotation = Quaternion.identity;
207  itemTransform.localScale = Vector3.one;
208  }

274  static public GameObject GetOrCreateCanvasGameObject()
275  {
276  GameObject selectedGo = Selection.activeGameObject;
277 
278  // Try to find a gameobject that is the selected GO or one if its parents.
279  Canvas canvas = (selectedGo != null) ? selectedGo.GetComponentInParent<Canvas>() : null;
280  if (canvas != null && canvas.gameObject.activeInHierarchy)
281  return canvas.gameObject;
282 
283  // No canvas in selection or its parents? Then use just any canvas..
284  canvas = Object.FindObjectOfType(typeof(Canvas)) as Canvas;
285  if (canvas != null && canvas.gameObject.activeInHierarchy)
286  return canvas.gameObject;
287 
288  // No canvas in the scene at all? Then create a new one.
