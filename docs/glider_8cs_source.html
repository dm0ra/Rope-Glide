312  private void SetNewPlayerVelocity()
313  {
314  // Area of wing
315  float area = 0.5f;
316  float airDensity = 1.225f;
317 
318  // Current velocity is weighted down to help glider function
319  float currVelo = this.player.GetComponent<Rigidbody2D>().velocity.magnitude;
320  currVelo = currVelo / 2.5f;
321 
322  // Angle of the glider
323  float tiltInAngles;
324  if ((this.glider.transform.eulerAngles.z > 0 && this.glider.transform.eulerAngles.z <= 61) || this.glider.transform.eulerAngles.z == 0)
325  {
326  tiltInAngles = this.glider.transform.eulerAngles.z;
327  }
328  else
329  {
330  tiltInAngles = this.glider.transform.eulerAngles.z - 360;
331  }
332 
333  float tiltInRads = 0.0174533f * tiltInAngles;
334 
335  // Get Coefficents
336  float liftCoefficent;
337  liftCoefficent = this.GetLiftCoefficent(tiltInAngles); // 2 * Mathf.PI * tiltInRads;
338  float dragCoefficent = this.GetDragCoefficent(tiltInAngles); // 1.28f * Mathf.Sin(tiltInRads);//
339 
340  // Calculate lift and drag forces
341  float lift = liftCoefficent * (currVelo * currVelo * airDensity / 2) * area;
342  float drag = dragCoefficent * (currVelo * currVelo * airDensity / 2) * area;
343  float weight = 9.8f * this.player.GetComponent<Rigidbody2D>().mass;
344  float stallSpeed = Mathf.Sqrt(2f * weight * 9.8f / (airDensity * area * (2 * Mathf.PI * 0.785398f)));
345 
346  // Calculate component forces of lift and drag.
347  float verticalLift = lift * Mathf.Cos(tiltInRads);
348  float horizontalLift = lift * Mathf.Sin(tiltInRads);
349  float verticalDrag = drag * Mathf.Sin(tiltInRads);
350  float horizontalDrag = drag * Mathf.Cos(tiltInRads);
351 
352  float balanceVertical = verticalLift + verticalDrag - weight;
353  float balanceHorizontal = horizontalLift - horizontalDrag;
354

356  float maxspeed = 300f;
357 
358  // Normal glider function less than max speend and greater than stall speed

361  Debug.Log("NORMALSPEED");
