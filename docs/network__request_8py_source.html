51 def unwrap_kwarg_namespace(func):
52  """Transform a Namespace object from argparse into proper args and kwargs.
53 
54  For a function that will be delegated to from argparse, inspect all of the
55  argments and extract them from the Namespace object.
56 
57  Args:
58  func: the function that we are wrapping to modify behavior
59 
60  Returns:
61  a new function that unwraps all of the arguments in a namespace and then
62  delegates to the passed function with those args.
63  """
64  # When we move to python 3, getfullargspec so that we can tell the
65  # difference between args and kwargs -- then this could be used for functions
66  # that have both args and kwargs
67  argspec = inspect.getargspec(func)
68 
69  def wrapped(argparse_namespace=None, **kwargs):
70  """Take a Namespace object and map it to kwargs.
71 
72  Inspect the argspec of the passed function. Loop over all the args that
73  are present in the function and try to map them by name to arguments in the
74  namespace. For keyword arguments, we do not require that they be present
75  in the Namespace.
76 
77  Args:
78  argparse_namespace: an arparse.Namespace object, the result of calling
79  argparse.ArgumentParser().parse_args()
80  **kwargs: keyword arguments that may be passed to the original function
81  Returns:
82  The return of the wrapped function from the parent.
83 
84  Raises:
85  ValueError in the event that an argument is passed to the cli that is not
86  in the set of named kwargs
87  """
88  if not argparse_namespace:
89  return func(**kwargs)
90 
91  reserved_namespace_keywords = ['func']
92  new_kwargs = {}
93 
94  args = argspec.args or []
95  for arg_name in args:
96  passed_value = getattr(argparse_namespace, arg_name, None)
97  if passed_value is not None:
98  new_kwargs[arg_name] = passed_value
99 
100  for namespace_key in vars(argparse_namespace).keys():
101  # ignore namespace keywords that have been set not passed in via cli
102  if namespace_key in reserved_namespace_keywords:
103  continue
104 
105  # make sure that we haven't passed something we should be processing
106  if namespace_key not in args:
107  raise ValueError('CLI argument "{}" does not match any argument in '
108  'function {}'.format(namespace_key, func.__name__))
109 
110  return func(**new_kwargs)
111 
112  wrapped.__name__ = func.__name__
113  return wrapped

123  def __init__(self, url, method, headers, body, timeout):
124  self.url = url.lower()
125  self.parsed_url = urlparse.urlparse(self.url)

132  def execute_request(self):
133  """"Execute the request, and get a response.
134 
135  Returns:
136  an HttpResponse object from httplib
137  """
138  if self.is_secure_connection:
139  conn = httplib.HTTPSConnection(self.get_hostname(), timeout=self.timeout)
140  else:
141  conn = httplib.HTTPConnection(self.get_hostname(), timeout=self.timeout)
142 
143  conn.request(self.method, self.url, self.body, self.headers)
144  response = conn.getresponse()
145  return response

171 def parse_colon_delimited_options(option_args):
172  """Parses a key value from a string.
173 
174  Args:
175  option_args: Key value string delimited by a color, ex: ("key:value")
176 
177  Returns:
178  Return an array with the key as the first element and value as the second
179 
180  Raises:
181  ValueError: If the key value option is not formatted correctly
182  """
183  options = {}
184 
185  if not option_args:
186  return options
187 
188  for single_arg in option_args:
189  values = single_arg.split(':')
190  if len(values) != 2:
191  raise ValueError('An option arg must be a single key/value pair '
192  'delimited by a colon - ex: "thing_key:thing_value"')
193 
194  key = values[0].strip()
195  value = values[1].strip()
196  options[key] = value
197 
198  return options

201 def make_request(request):
202  """Makes a synchronous network request and return the HTTP status code.
203 
204  Args:
205  request: a well formulated request object
206 
207  Returns:
208  The HTTP status code of the network request.
209  '1' maps to invalid request headers.
210  """
211 
212  logger.info('Sending network request -')
213  logger.info('\tUrl: %s', request.url)
214  logger.debug('\tMethod: %s', request.method)
215  logger.debug('\tHeaders: %s', request.headers)
216  logger.debug('\tBody: %s', request.body)
217 
218  try:
219  response = request.execute_request()
220  except socket.timeout:
221  logger.exception(
222  'Timed out post request to %s in %d seconds for request body: %s',
223  request.url, request.timeout, request.body)
224  return EXIT_CODE_HTTP_TIMEOUT
225  except (httplib.HTTPException, socket.error):
226  logger.exception(
227  'Encountered generic exception in posting to %s with request body %s',
228  request.url, request.body)
229  return EXIT_CODE_GENERIC_HTTPLIB_ERROR
230 
231  status = response.status
232  headers = response.getheaders()
233  logger.info('Received Network response -')
234  logger.info('\tStatus code: %d', status)
235  logger.debug('\tResponse headers: %s', headers)
236 
237  if status < 200 or status > 299:
238  logger.error('Request (%s) failed with status code %d\n', request.url,
239  status)
240 
241  # If we wanted this script to support get, we need to
242  # figure out what mechanism we intend for capturing the response
243  return status

333 def map_http_status_to_exit_code(status_code):
334  """Map an http status code to the appropriate exit code.
335 
336  Exit codes in python are valid from 0-256, so we want to map these to
337  predictable exit codes within range.
338 
339  Args:
340  status_code: the input status code that was output from the network call
341  function
342 
343  Returns:
344  One of our valid exit codes declared at the top of the file or a generic
345  unknown error code
346  """
347  if status_code <= MAX_EXIT_CODE:
348  return status_code
349 
350  if status_code > 199 and status_code < 300:
351  return EXIT_CODE_SUCCESS
352 
353  if status_code == 302:
354  return EXIT_CODE_HTTP_REDIRECT_ERROR
355 
356  if status_code == 404:
357  return EXIT_CODE_HTTP_NOT_FOUND_ERROR
358 
359  if status_code > 499:
360  return EXIT_CODE_HTTP_SERVER_ERROR
361 
362  return EXIT_CODE_HTTP_UNKNOWN_ERROR
