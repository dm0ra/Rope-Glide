388  public bool IsHingeClosest(int index)
389  {
390  bool ret = true;
391  float currDistance; // current distance
392  float distanceInQuestion; // distance of hinge in question
393  Vector2 nextVector; // next hinge to look at
394  Vector2 playerVector = new Vector2(this.Player.transform.position.x, this.Player.transform.position.y); // gets player 2d vector
395  Vector2 hingeVector = new Vector2(this.Hinges[index].transform.position.x, this.Hinges[index].transform.position.y); // gets the current hinge 2d vector
396 
397  distanceInQuestion = Vector2.Distance(playerVector, hingeVector); // finds the distance between the current hinge and the play
398 
399  // loops through all the hinges
400  for (int x = 0; x < this.Hinges.Count; x++)
401  {
402  // makes sure not to check the current hinge in question
403  if (x != index)
404  {
405  nextVector = new Vector2(this.Hinges[x].transform.position.x, this.Hinges[x].transform.position.y);
406  currDistance = Vector2.Distance(playerVector, nextVector);
407 
408  if (distanceInQuestion < currDistance)
409  {
410  continue;
411  }
412  else
413  {
414  ret = false;
415  break;
416  }
417  }
418  }
419 
420  return ret;
421  }
