293  public bool IsHingeClosest(int index)
294  {
295  bool ret = true;
296  float currDistance; // current distance
297  float distanceInQuestion; // distance of hinge in question
298  Vector2 nextVector; // next hinge to look at
299  Vector2 playerVector = new Vector2(this.player.transform.position.x, this.player.transform.position.y); // gets player 2d vector
300  Vector2 hingeVector = new Vector2(this.hinges[index].transform.position.x, this.hinges[index].transform.position.y); // gets the current hinge 2d vector
301 
302  distanceInQuestion = Vector2.Distance(playerVector, hingeVector); // finds the distance between the current hinge and the play
303 
304  // loops through all the hinges
305  for (int x = 0; x < this.hinges.Count; x++)
306  {
307  // makes sure not to check the current hinge in question
308  if (x != index)
309  {
310  nextVector = new Vector2(this.hinges[x].transform.position.x, this.hinges[x].transform.position.y);
311  currDistance = Vector2.Distance(playerVector, nextVector);
312 
313  if (distanceInQuestion < currDistance)
314  {
315  continue;
316  }
317  else
318  {
319  ret = false;
320  break;
321  }
322  }
323  }
324 
325  return ret;
