45  // Gets an array of sorting layer names.
46  // Since this uses reflection, callers should check for 'null' which will be returned if the reflection fails.
47  public static string[] sortingLayerNames
48  {
49  get
50  {
51  if (_sortingLayerNamesProperty == null)
52  {
53  return null;
54  }
55 
56  return _sortingLayerNamesProperty.GetValue(null, null) as string[];
57  }
58  }

60  // Given the ID of a sorting layer, returns the sorting layer's name
61  public static string GetSortingLayerNameFromID(int id)
62  {
63  string[] names = sortingLayerNames;
64  if (names == null)
65  {
66  return null;
67  }
68 
69  for (int i = 0; i < names.Length; i++)
70  {
71  if (GetSortingLayerIDForIndex(i) == id)
72  {
73  return names[i];
74  }
75  }
76 
77  return null;
78  }
79 
80  // Given the name of a sorting layer, returns the ID.
81  public static int GetSortingLayerIDForName(string name)
82  {
83  string[] names = sortingLayerNames;
84  if (names == null)
85  {
86  return 0;
87  }
88 
89  return GetSortingLayerIDForIndex(Array.IndexOf(names, name));
90  }
91 
92  // Helper to convert from a sorting layer INDEX to a sorting layer ID. These are not the same thing.
93  // IDs are based on the order in which layers were created and do not change when reordering the layers.
94  // Thankfully there is a private helper we can call to get the ID for a layer given its index.
95  public static int GetSortingLayerIDForIndex(int index)
96  {
97  if (_getSortingLayerUserIdMethod == null)
98  {
99  return 0;
100  }
101 
102  return (int)_getSortingLayerUserIdMethod.Invoke(null, new object[] { index });
103  }
104  }
